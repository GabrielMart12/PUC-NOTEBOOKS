SELECT B.Nome, COUNT(*) AS TOT, AVG(A.Horas) AS MEDIA, SUM(A.Custo) AS SOMA
FROM CURSO AS A
INNER JOIN AREA AS B ON (AREA.Sigla = CURSO.Area)
INNER JOIN (
	   SELECT AREA.Sigla
	   FROM AREA
	   EXCEPT
	   SELECT AREA.SuperAREA
	   FROM AREA
	   ) AS C
ON (B.Sigla = C.Sigla)
GROUP BY B.Nome
